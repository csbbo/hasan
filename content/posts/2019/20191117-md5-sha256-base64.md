<div id="app">
    <span>base64:</span><input class="input" v-model="base64">
    <p class="base64TagP">{{base64Out}}</p><br>
    
    <span>sha256:</span><input class="input" v-model="sha256">

<script>
    var app = new Vue({
        el: '#app',
        data: {
            md5: "",
            sha256: "",
            base64: "",
        },
        computed:{
            md5Out: function () {
                if(this.md5==""){
                    return ""
                }
                return md5(this.md5)
            },
            sha256Out: function() {
                if(this.sha256==""){
                    return ""
                }
                return sha256(this.sha256)
            },
            base64Out: function() {
                return btoa(this.base64)
            }
        }
    })
</script>
